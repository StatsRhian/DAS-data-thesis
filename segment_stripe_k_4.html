library(mclust)
library(rARPACK)
library(animation)
library(streamMOA)
library(rJava)
library(stream)
library(proxy)
ani.options(interval = 0.2, loop = FALSE)
for (t in (sizeInit + 1):runs) {
    if (t%%batchSize == 0) {
        update(clustream, stream, batchSize)
        assignment = as.vector(get_assignment(clustream, 
            data[1:(t + sizeInit), ], type = "macro"))
        segment_lines = which(diff(assignment) != 0)[rle(assignment)$lengths > 
            20]
        segment_lines = segment_lines[-c(1, length(segment_lines))]
        par(mfrow = c(2, 3), las = 1)
        for (i in c(1:nDim)) {
            plot(1:N, data[, i], main = i, type = "l")
            abline(v = (t + sizeInit), col = 4)
            abline(v = segment_lines, col = 2)
        }
    }
}
